select concat(utilisateur.nom," ",utilisateur.prenom) as Utilisateur,plainte as Plainte,villecentre as Ville,intitule as Intitule,atout as Atouts,satisfaction as Satisfaction from reponse inner join utilisateur on reponse.idUtilisateur=utilisateur.idUtilisateur inner join plaintes on reponse.idPlainte=plaintes.idPlainte inner join villecentre on reponse.idVillecentre=villecentre.idVillecentre inner join intituleforma on reponse.idIntitule=intituleforma.idIntitule inner join atouts on reponse.idAtouts=atouts.idAtouts inner join satisfaction on reponse.idSatisf=satisfaction.idSatisf;
